Google Sheets Connection

One-time setup (~3 min):
  1. Create a new Google Sheet (keep it private — don't share it with anyone)
  2. In the sheet, go to Extensions → Apps Script
  3. Delete any code there and paste the script below
  4. Important: In the script, change YOUR_SECRET_HERE to a passphrase of your choice
  5. Click Deploy → New deployment → Web app → Execute as Me → Who has access: Anyone → Deploy
  6. Copy the deployment URL and paste it above, along with the same passphrase

The sheet stays fully private. The Apps Script runs as your account so it can access your sheet. "Anyone" access means anyone can call the script URL, but the passphrase blocks unauthorized requests. No API key or Google Cloud project needed.

Apps Script code (click to expand & copy) 
// ⚠️ Change this to your own secret passphrase
var SECRET = "YOUR_SECRET_HERE";

function doPost(e) {
  var raw = e.postData.contents;
  var data;

  // Handle both JSON and form-encoded (payload=JSON) formats
  try {
    data = JSON.parse(raw);
  } catch(err) {
    // Try form-encoded: payload={json}
    var params = {};
    raw.split("&").forEach(function(pair) {
      var kv = pair.split("=");
      params[decodeURIComponent(kv[0])] = decodeURIComponent(kv[1] || "");
    });
    if (params.payload) {
      data = JSON.parse(params.payload);
    } else {
      return ContentService
        .createTextOutput(JSON.stringify({error: "invalid request format"}))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }

  if (data.passphrase !== SECRET) {
    return ContentService
      .createTextOutput(JSON.stringify({error: "unauthorized"}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Sheet1");

  if (data.action === "read") {
    var all = sheet.getDataRange().getValues();
    var meta = PropertiesService.getScriptProperties().getProperty("meta") || "{}";
    return ContentService
      .createTextOutput(JSON.stringify({success: true, values: all, meta: meta}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // Write action: ONLY proceed if rows are provided
  if (!data.rows || data.rows.length === 0) {
    return ContentService
      .createTextOutput(JSON.stringify({error: "no rows provided, refusing to clear sheet"}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  sheet.clearContents();
  sheet.getRange(1, 1, 1, 6).setValues(
    [["Date","Description","Category","Card","Amount","ID"]]
  );
  sheet.getRange(2, 1, data.rows.length, 6).setValues(data.rows);

  if (data.meta) {
    PropertiesService.getScriptProperties().setProperty("meta", data.meta);
  }

  return ContentService
    .createTextOutput(JSON.stringify({success: true, count: data.rows.length}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  // If ANY query parameters exist, treat as a read request
  var params = e.parameter || {};
  var hasParams = Object.keys(params).length > 0;

  if (hasParams) {
    var passphrase = params.passphrase || params.p || "";
    if (passphrase !== SECRET) {
      return ContentService
        .createTextOutput(JSON.stringify({error: "unauthorized"}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1");
    var all = sheet.getDataRange().getValues();
    var meta = PropertiesService.getScriptProperties().getProperty("meta") || "{}";
    return ContentService
      .createTextOutput(JSON.stringify({success: true, values: all, meta: meta}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // No params = connection test
  return ContentService
    .createTextOutput(JSON.stringify({status: "ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
